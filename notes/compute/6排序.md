### 排序算法的分析
- 排序算法的执行效率
  - 最好、最坏、平均时间复杂度
  - 同一时间复杂度的系数、常亮、低阶
  - 比价次数和交换（或移动）次数
- 排序算法的内存消耗
- 排序算法的稳定性
```js
// 冒泡排序（原地、文档算法）
function bubbleSort(a, n) { // a表示数组，n表示数组大小
  if (n <= 1) return;
 
 for (var i = 0; i < n; ++i) {
    // 提前退出冒泡循环的标志位
    var flag = false;
    for (var j = 0; j < n - i - 1; ++j) {
      if (a[j] > a[j+1]) { // 交换
        var tmp = a[j];
        a[j] = a[j+1];
        a[j+1] = tmp;
        flag = true;  // 表示有数据交换      
      }
    }
    if (!flag) break;  // 没有数据交换，提前退出
  }
}

// 插入排序（原地、文档算法，比冒泡更高效）
function insertionSort(a, n) { // a表示数组，n表示数组大小
  if (n <= 1) return;

  for (var i = 1; i < n; ++i) {
    var value = a[i];
    var j = i - 1;
    // 查找插入的位置
    for (; j >= 0; --j) {
      if (a[j] > value) {
        a[j+1] = a[j];  // 数据移动
      } else {
        break;
      }
    }
    a[j+1] = value; // 插入数据
  }
}
// 选择排序（不稳地算法）
```
