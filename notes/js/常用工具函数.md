## 常用工具函数
### 生成一个随机boolean值
```js
const randomBoolean = () => Math.random() >= 0.5;
```
### 检查一个日期是否为工作日
```js
const isWeekday = (date) => date.getDay() % 6 !== 0;
```
### 反转字符串
```js
const reverse = str => str.split('').reverse().join('');
```
### 检查当前元素是否在试图内或聚焦
```js
const isBrowserTabInView = () => document.hidden;
```
### 判断数字奇偶
```js
isEven = num => num % 2 === 0;
```
### 截取固定小数点的数字
```js
const toFixed = (n, fixed) => ~~(Math.pow(10, fixed) * n) / Math.pow(10, fixed);
// Examples
toFixed(25.198726354, 1);       // 25.1
toFixed(25.198726354, 2);       // 25.19
```
### 检查一个元素是否聚焦
```js
const elementIsInFocus = (el) => (el === document.activeElement);
```
### 检查是否支持触摸事件
```js
const touchSupported = () => {
  ('ontouchstart' in window || window.DocumentTouch && document instanceof window.DocumentTouch);
}
```
### 检查设备
```js
// 是否为iphone类设备
const isAppleDevice = /Mac|iPod|iPhone|iPad/.test(navigator.platform);
// 是否为微信浏览器
const isWechar = /micromessenger/.test(navigator.userAgent.toLowerCase());
```
### 华氏度和摄氏度的抓换
```js
// 摄氏度转华氏度
const celsiusToFahrenheit = (celsius) => celsius * 9/5 + 32;
// 华氏度转摄氏度
const fahrenheitToCelsius = (fahrenheit) => (fahrenheit - 32) * 5/9;
```
### 数字千分位展示
```js
const thusandthPrice = (val) => {
  (val && String(val).replace(/\B(?=(\d{3})+(?!\d))/g, ','))
}
```
### 小些数字转换大写
```js
function digitUppercase(n: number): string {
  var fraction = ['角', '分'];
  var digit = ['零', '壹', '贰', '叁', '肆', '伍', '陆', '柒', '捌', '玖'];
  var unit = [
    ['元', '万', '亿'],
    ['', '拾', '佰', '仟']
  ];
  var head = n < 0 ? '欠' : '';
  n = Math.abs(n);
  var s = '';
  for (var i = 0; i < fraction.length; i++) {
    s += (
      digit[Math.floor(n * 10 * Math.pow(10, i)) % 10] + fraction[i]
    ).replace(/零./, '');
  }
  s = s || '整';
  n = Math.floor(n);
  for (var i = 0; i < unit[0].length && n > 0; i++) {
    var p = '';
    for (var j = 0; j < unit[1].length && n > 0; j++) {
      p = digit[n % 10] + unit[1][j] + p;
      n = Math.floor(n / 10);
    }
    s = p.replace(/(零.)*零$/, '').replace(/^$/, '零') + unit[0][i] + s;
  }
  return (
    head +
    s
      .replace(/(零.)*零元/, '元')
      .replace(/(零.)+/g, '零')
      .replace(/^整$/, '零元整')
  );
}
```
### 跨域图片下载（a标签download属性不生效时）
```js
const downloadByLink = (name: string, data: Blob) => {
  const urlObject = window.URL || window.webkitURL || window;
  // eslint-disable-next-line
  const export_blob = new Blob([data]);
  const link = document.createElement('a');
  link.href = urlObject.createObjectURL(export_blob);
  link.download = name;
  link.click();
};
const downloadUrlFile = (url: string, name: string) => {
  const ajax = new XMLHttpRequest();
  ajax.open('GET', url, true);
  ajax.responseType = 'blob';
  ajax.onload = (e: any) => {
    downloadByLink(name, e.target.response);
  };
  ajax.send();
};
// Example
downloadUrlFile(url, downloadFileName);
```
